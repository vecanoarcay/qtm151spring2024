# Add a shaded region wth a rectangle
# "x" is the x-coordinates, "y1" and "y2" are the lower
# and upper bounds of the rectangle. We can set this
# to be the minimum and meximum of the outcome.
# we use "where" to test a logical condition

#creating boolean vector
vec_bool = exchangerates["usd_to_jpy"] < 100

vec_y =  exchangerates["usd_to_jpy"]
plt.fill_between(x= exchangerates["date"],
                 y1 = vec_y.min(),
                 y2 = vec_y.max(),
                 where = vec_bool,
                 alpha = 0.2,color = "red")

plt.show()

month_config = pd.Grouper(key='date', freq='y')

monthlydata = (exchangerates
               .groupby(month_config) 
               .agg(usd_to_rmb_std = ("usd_to_rmb","std"))
               .reset_index())
